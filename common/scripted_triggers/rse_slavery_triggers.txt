# Broad permission for concubinage — not just Islamic
rse_concubines_permitted_trigger = {
	OR = {
		faith = { religion_tag = islam_religion }
		faith = { has_doctrine_parameter = unrestricted_concubines }
		culture = { has_cultural_tradition = tradition_concubines }
		faith = { religion_tag = germanic_religion }
		faith = { religion_tag = tengri_religion }
		faith = { religion_tag = west_african_religion }
		faith = { religion_tag = african_religion }
		faith = { religion_tag = magyar_religion }
	}
}

# Broad permission for slavery — Islamic, pagan, raider-culture, or Christian against non-Christians
rse_thralldom_permitted_trigger = {
	OR = {
		faith = { religion_tag = islam_religion }
		faith = { religion_tag = germanic_religion }
		faith = { religion_tag = tengri_religion }
		faith = { religion_tag = west_african_religion }
		faith = { religion_tag = african_religion }
		faith = { religion_tag = magyar_religion }
		faith = { religion_tag = baltic_religion }
		faith = { religion_tag = finno_ugric_religion }
		faith = { religion_tag = slavic_religion }
		faith = { religion_tag = sami_religion }
		faith = { religion_tag = north_eurasian_religion }
		faith = { religion_tag = zun_religion }
		culture = { has_cultural_tradition = tradition_chosen_warriors }
		culture = { has_cultural_tradition = tradition_warrior_culture }
		culture = { has_cultural_tradition = tradition_horse_lords }
		culture = { has_cultural_tradition = tradition_seafaring }
		culture = { has_cultural_tradition = tradition_steppe_tolerance }
		has_trait = rse_slaver_apprentice
		has_trait = rse_slave_broker
		has_trait = rse_taskmaster
		has_trait = rse_grand_slaver
		has_trait = rse_mamluk_lord
		has_trait = rse_harem_keeper
		AND = {
			faith = { religion_tag = christianity_religion }
			rse_christian_slavery_target_exists_trigger = yes
		}
	}
}

# Christian rulers may practice slavery only if non-Christian targets exist
rse_christian_slavery_target_exists_trigger = {
	OR = {
		any_courtier = {
			NOT = { faith = { religion_tag = christianity_religion } }
		}
		any_held_county = {
			NOT = { faith = { religion_tag = christianity_religion } }
		}
		any_neighboring_top_liege_realm_owner = {
			NOT = { faith = { religion_tag = christianity_religion } }
		}
		any_neighboring_top_liege_realm_owner = {
			culture = { has_cultural_pillar = heritage_west_slavic }
		}
		any_neighboring_top_liege_realm_owner = {
			culture = { has_cultural_pillar = heritage_east_slavic }
		}
	}
}

rse_is_orthodox_trigger = {
	faith = {
		OR = {
			this = faith:orthodox
			has_doctrine = tenet_iconography
		}
	}
}

rse_is_sunni_trigger = {
	faith = {
		OR = {
			this = faith:ashari
			this = faith:maturidi
			this = faith:hanbali
			this = faith:mutazila
		}
	}
}

# A character may be enslaved only if they are NOT a member of a noble dynasty.
# Lowborn folk, freshly-created captives and commoners can be taken; nobles cannot.
# A character may be enslaved if they are lowborn OR a member of an
# obscure (low-prestige) dynasty. Members of renowned dynasties — those
# whose dynasty has risen to splendour — are protected from enslavement.
rse_can_be_enslaved_trigger = {
	NOT = { has_trait = rse_enslaved }
	OR = {
		is_lowborn = yes
		AND = {
			exists = dynasty
			dynasty = {
				dynasty_prestige_level <= 1
			}
		}
	}
}
