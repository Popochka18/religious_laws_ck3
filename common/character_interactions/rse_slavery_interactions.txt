rse_enslave_prisoner_interaction = {
	category = interaction_category_prison

	common_interaction = yes
	interface_priority = 100

	desc = rse_enslave_prisoner_interaction_desc

	is_shown = {
		scope:actor = {
			is_landed = yes
			is_adult = yes
		}
		scope:recipient = {
			is_imprisoned_by = scope:actor
			rse_can_be_enslaved_trigger = yes
		}
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			NOT = { has_trait = rse_chainbreaker }
			NOT = { has_trait = rse_manumitter }
		}
	}

	on_accept = {
		scope:recipient = {
			release_from_prison = yes
			move_to_pool = yes
			set_employer = scope:actor
			add_trait = rse_enslaved
			add_opinion = {
				modifier = enslaved_opinion
				target = scope:actor
				years = 100
			}
		}
		scope:actor = {
			add_gold = 25
			add_piety = minor_piety_loss
		}
	}

	auto_accept = { always = yes }

	ai_frequency = 0

	ai_potential = {
		exists = this
		is_ai = yes
	}

	ai_will_do = {
		base = 0
	}
}

rse_raid_for_slaves_interaction = {
	category = interaction_category_hostile

	common_interaction = yes
	interface_priority = 90

	desc = rse_raid_for_slaves_interaction_desc

	is_shown = {
		scope:actor = {
			is_landed = yes
			is_adult = yes
			is_at_war_with = scope:recipient
		}
		scope:recipient = {
			is_landed = yes
			NOT = { this = scope:actor }
		}
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			is_at_war_with = scope:recipient
		}
	}

	cooldown = { years = 3 }

	on_accept = {
		scope:recipient = {
			random_held_county = {
				save_scope_as = raided_county
				add_county_modifier = {
					modifier = slaves_seized_modifier
					years = 5
				}
				change_development_level = -1
			}
		}
		scope:actor = {
			add_gold = 150
			add_prestige = -25
			add_piety = minor_piety_loss
			create_character = {
				template_character = scope:actor
				location = scope:actor.capital_province
				faith = scope:raided_county.faith
				culture = scope:raided_county.culture
				gender_female_chance = 50
				save_scope_as = slave_captive_1
				employer = scope:actor
			}
			scope:slave_captive_1 = {
				add_trait = rse_enslaved
				add_opinion = {
					modifier = enslaved_opinion
					target = scope:actor
					years = 100
				}
			}
			create_character = {
				template_character = scope:actor
				location = scope:actor.capital_province
				faith = scope:raided_county.faith
				culture = scope:raided_county.culture
				gender_female_chance = 50
				save_scope_as = slave_captive_2
				employer = scope:actor
			}
			scope:slave_captive_2 = {
				add_trait = rse_enslaved
				add_opinion = {
					modifier = enslaved_opinion
					target = scope:actor
					years = 100
				}
			}
		}
		scope:recipient = {
			trigger_event = { id = rse_slave.0009 days = { 1 5 } }
		}
	}

	auto_accept = { always = yes }

	ai_frequency = 0

	ai_potential = {
		exists = this
		is_ai = yes
	}

	ai_will_do = {
		base = 0
	}
}
