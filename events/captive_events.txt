# First days in captivity — the captor's tone is set
rse_captive.0001 = {
	type = character_event
	title = rse_captive.0001.t
	desc = rse_captive.0001.desc
	theme = faith

	left_portrait = {
		character = root
		animation = shame
	}
	right_portrait = {
		character = scope:captor
		animation = personality_bold
	}

	trigger = {
		is_imprisoned = yes
	}

	immediate = {
		every_living_character = {
			limit = { any_prisoner = { this = root } }
			save_scope_as = captor
		}
	}

	option = {
		name = rse_captive.0001.a
		trigger = { gold >= 50 }
		add_gold = -50
		add_character_modifier = {
			modifier = bribed_jailers_modifier
			years = 3
		}
	}

	option = {
		name = rse_captive.0001.b
		add_character_modifier = {
			modifier = stoic_in_chains_modifier
			years = 3
		}
		add_piety = minor_piety_gain
		stress_impact = { brave = miniscule_stress_impact_loss }
	}

	option = {
		name = rse_captive.0001.c
		scope:captor = {
			add_opinion = {
				modifier = groveling_prisoner_opinion
				target = root
				years = 10
			}
		}
		add_prestige = minor_prestige_loss
		stress_impact = { arrogant = major_stress_impact_gain }
	}
}

# A cellmate proposes a plot
rse_captive.0006 = {
	type = character_event
	title = rse_captive.0006.t
	desc = rse_captive.0006.desc
	theme = faith

	left_portrait = root
	right_portrait = {
		character = scope:cellmate
		animation = personality_bold
	}

	trigger = {
		is_imprisoned = yes
		is_adult = yes
	}

	immediate = {
		every_living_character = {
			limit = { any_prisoner = { this = root } }
			save_scope_as = captor
		}
		scope:captor = {
			random_prisoner = {
				limit = {
					NOT = { this = root }
					is_adult = yes
				}
				save_scope_as = cellmate
			}
		}
		if = {
			limit = { NOT = { exists = scope:cellmate } }
			create_character = {
				template_character = root
				location = root.location
				faith = root.faith
				culture = root.culture
				save_scope_as = cellmate
				random_traits_list = {
					count = 1
					50 = { add_trait = brave }
					50 = { add_trait = deceitful }
					50 = { add_trait = ambitious }
				}
			}
			scope:cellmate = {
				imprison = { target = scope:captor type = dungeon }
			}
		}
	}

	option = {
		name = rse_captive.0006.a
		random = {
			chance = 50
			release_from_prison = yes
			scope:cellmate = {
				release_from_prison = yes
				add_opinion = {
					modifier = grateful_opinion
					target = root
					years = 50
				}
			}
		}
		random = {
			chance = 50
			add_character_modifier = {
				modifier = failed_escape_modifier
				years = 3
			}
		}
		add_prestige = minor_prestige_gain
	}

	option = {
		name = rse_captive.0006.b
		add_character_modifier = {
			modifier = bribed_jailers_modifier
			years = 2
		}
	}

	option = {
		name = rse_captive.0006.c
		scope:captor = {
			add_opinion = {
				modifier = useful_prisoner_opinion
				target = root
				years = 15
			}
		}
		scope:cellmate = {
			death = { death_reason = death_execution killer = scope:captor }
		}
		add_dread = minor_dread_gain
		stress_impact = { just = major_stress_impact_gain compassionate = major_stress_impact_gain }
	}
}

# Captor threatens to sell into slavery
rse_captive.0007 = {
	type = character_event
	title = rse_captive.0007.t
	desc = rse_captive.0007.desc
	theme = faith

	left_portrait = {
		character = root
		animation = fear
	}
	right_portrait = {
		character = scope:captor
		animation = personality_greedy
	}

	trigger = {
		is_imprisoned = yes
	}

	immediate = {
		every_living_character = {
			limit = {
				any_prisoner = { this = root }
				OR = {
					has_trait = rse_slaver_apprentice
					has_trait = rse_slave_broker
					has_trait = rse_taskmaster
					has_trait = rse_grand_slaver
					has_trait = greedy
				}
			}
			save_scope_as = captor
		}
	}

	option = {
		name = rse_captive.0007.a
		trigger = { gold >= 150 }
		add_gold = -150
		release_from_prison = yes
		add_prestige = minor_prestige_loss
	}

	option = {
		name = rse_captive.0007.b
		add_trait = rse_enslaved
		release_from_prison = yes

		set_employer = scope:captor
		add_opinion = {
			modifier = enslaved_opinion

		}

	option = {
