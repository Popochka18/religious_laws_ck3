# A slave bears a child — fate decided by the owner
rse_owner.0001 = {
	type = character_event
	title = rse_owner.0001.t
	desc = rse_owner.0001.desc
	theme = faith

	left_portrait = root
	right_portrait = {
		character = scope:slave_mother
		animation = personality_compassionate
	}

	trigger = {
		is_landed = yes
		is_adult = yes
		any_courtier = {
			has_trait = rse_enslaved
			is_female = yes
			is_adult = yes
		}
	}

	immediate = {
		random_courtier = {
			limit = {
				has_trait = rse_enslaved
				is_female = yes
				is_adult = yes
			}
			save_scope_as = slave_mother
		}
		create_character = {
			template_character = root
			location = root.capital_province
			faith = scope:slave_mother.faith
			culture = scope:slave_mother.culture
			gender_female_chance = 50
			age = 0
			save_scope_as = slave_child
			employer = root
			mother = scope:slave_mother
		}
		scope:slave_child = { add_trait = rse_enslaved }
	}

	option = {
		name = rse_owner.0001.a
		scope:slave_mother = {
			add_opinion = {
				modifier = grateful_opinion
				target = root
				years = 50
			}
		}
		add_piety = minor_piety_loss
	}

	option = {
		name = rse_owner.0001.b
		scope:slave_mother = {
			remove_trait = rse_enslaved
			add_trait = rse_freed_slave
			add_opinion = {
				modifier = grateful_opinion
				target = root
				years = 100
			}
		}
		scope:slave_child = {
			remove_trait = rse_enslaved
			add_trait = rse_freed_slave
		}
		add_piety = medium_piety_gain
		add_gold = -50
	}

	option = {
		name = rse_owner.0001.c
		scope:slave_child = { death = { death_reason = death_disappearance } }
		add_dread = medium_dread_gain
		scope:slave_mother = {
			add_opinion = {
				modifier = scorned_opinion
				target = root
				years = 100
			}
		}
		stress_impact = {
			compassionate = major_stress_impact_gain
			just = major_stress_impact_gain
		}
	}
}
