namespace = rse_owner

# A slave bears a child — fate decided by the owner
rse_owner.0001 = {
	type = character_event
	title = rse_owner.0001.t
	desc = rse_owner.0001.desc
	theme = faith

	left_portrait = root
	right_portrait = {
		character = scope:slave_mother
		animation = personality_compassionate
	}

	trigger = {
		is_landed = yes
		is_adult = yes
		any_courtier = {
			has_trait = rse_enslaved
			is_female = yes
			is_adult = yes
		}
	}

	immediate = {
		random_courtier = {
			limit = {
				has_trait = rse_enslaved
				is_female = yes
				is_adult = yes
			}
			save_scope_as = slave_mother
		}
		create_character = {
			template_character = root
			location = root.capital_province
			faith = scope:slave_mother.faith
			culture = scope:slave_mother.culture
			gender_female_chance = 50
			age = 0
			save_scope_as = slave_child
			employer = root
			mother = scope:slave_mother
		}
		scope:slave_child = { add_trait = rse_enslaved }
	}

	option = {
		name = rse_owner.0001.a
		scope:slave_mother = {
			add_opinion = {
				modifier = grateful_opinion
				target = root
				years = 50
			}
		}
		add_piety = minor_piety_loss
	}

	option = {
		name = rse_owner.0001.b
		scope:slave_mother = {
			remove_trait = rse_enslaved
			add_trait = rse_freed_slave
			add_opinion = {
				modifier = grateful_opinion
				target = root
				years = 100
			}
		}
		scope:slave_child = {
			remove_trait = rse_enslaved
			add_trait = rse_freed_slave
		}
		add_piety = medium_piety_gain
		add_gold = -50
	}

	option = {
		name = rse_owner.0001.c
		scope:slave_child = { death = { death_reason = death_disappearance } }
		add_dread = medium_dread_gain
		scope:slave_mother = {
			add_opinion = {
				modifier = scorned_opinion
				target = root
				years = 100
			}
		}
		stress_impact = {
			compassionate = major_stress_impact_gain
			just = major_stress_impact_gain
		}
	}
}

# Two slaves are in love
rse_owner.0002 = {
	type = character_event
	title = rse_owner.0002.t
	desc = rse_owner.0002.desc
	theme = faith

	left_portrait = root
	right_portrait = {
		character = scope:slave_lover_a
		animation = personality_compassionate
	}

	trigger = {
		is_landed = yes
		any_courtier = {
			has_trait = rse_enslaved
			is_adult = yes
			is_female = yes
		}
		any_courtier = {
			has_trait = rse_enslaved
			is_adult = yes
			is_male = yes
		}
	}

	immediate = {
		random_courtier = {
			limit = {
				has_trait = rse_enslaved
				is_adult = yes
				is_female = yes
			}
			save_scope_as = slave_lover_a
		}
		random_courtier = {
			limit = {
				has_trait = rse_enslaved
				is_adult = yes
				is_male = yes
			}
			save_scope_as = slave_lover_b
		}
	}

	option = {
		name = rse_owner.0002.a
		scope:slave_lover_a = { marry = scope:slave_lover_b }
		scope:slave_lover_a = {
			add_opinion = {
				modifier = grateful_opinion
				target = root
				years = 50
			}
		}
		scope:slave_lover_b = {
			add_opinion = {
				modifier = grateful_opinion
				target = root
				years = 50
			}
		}
		add_piety = minor_piety_gain
	}

	option = {
		name = rse_owner.0002.b
		scope:slave_lover_b = {
			set_employer = liege
			banish = yes
		}
		scope:slave_lover_a = {
			add_opinion = {
				modifier = scorned_opinion
				target = root
				years = 30
			}
		}
		add_gold = 50
	}

	option = {
		name = rse_owner.0002.c
		scope:slave_lover_a = {
			death = { death_reason = death_execution killer = root }
		}
		scope:slave_lover_b = {
			death = { death_reason = death_execution killer = root }
		}
		add_dread = major_dread_gain
		add_piety = minor_piety_gain
		stress_impact = { compassionate = major_stress_impact_gain }
	}
}

# A slave teaches the children well
rse_owner.0008 = {
	type = character_event
	title = rse_owner.0008.t
	desc = rse_owner.0008.desc
	theme = faith

	left_portrait = root
	right_portrait = {
		character = scope:teacher_slave
		animation = personality_compassionate
	}

	trigger = {
		is_landed = yes
		any_courtier = {
			has_trait = rse_enslaved
			learning >= 8
		}
		any_child = {
			is_alive = yes
			NOT = { is_adult = yes }
		}
	}

	immediate = {
		random_courtier = {
			limit = {
				has_trait = rse_enslaved
				learning >= 8
			}
			save_scope_as = teacher_slave
		}
		random_child = {
			limit = {
				is_alive = yes
				NOT = { is_adult = yes }
			}
			save_scope_as = bright_child
		}
	}

	option = {
		name = rse_owner.0008.a
		scope:teacher_slave = {
			remove_trait = rse_enslaved
			add_trait = rse_freed_slave
			add_opinion = {
				modifier = grateful_opinion
				target = root
				years = 100
			}
		}
		add_piety = medium_piety_gain
		scope:bright_child = {
			add_opinion = {
				modifier = secret_friend_opinion
				target = scope:teacher_slave
				years = 100
			}
		}
	}

	option = {
		name = rse_owner.0008.b
		scope:teacher_slave = {
			add_trait = rse_house_slave
			add_opinion = {
				modifier = grateful_opinion
				target = root
				years = 30
			}
		}
		add_gold = -25
	}

	option = {
		name = rse_owner.0008.c
		scope:teacher_slave = {
			set_employer = liege
		}
		add_gold = 150
		scope:bright_child = {
			stress_impact = { compassionate = medium_stress_impact_gain }
		}
	}
}
