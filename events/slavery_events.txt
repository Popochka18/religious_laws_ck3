namespace = rse_slave

# A slave market opens in your domain
rse_slave.0001 = {
	type = character_event
	title = rse_slave.0001.t
	desc = rse_slave.0001.desc
	theme = faith

	left_portrait = root
	right_portrait = {
		character = scope:slave_trader
		animation = personality_greedy
	}

	trigger = {
		is_landed = yes
		any_held_county = { count >= 1 }
		NOT = { has_character_flag = rse_slave_market_decided }
	}

	immediate = {
		random_held_county = { save_scope_as = market_county }
		create_character = {
			template_character = root
			location = scope:market_county.title_province
			faith = root.faith
			culture = root.culture
			save_scope_as = slave_trader
		}
	}

	option = {
		name = rse_slave.0001.a
		add_character_flag = rse_slave_market_decided
		scope:market_county = {
			add_county_modifier = {
				modifier = slave_market_modifier
				years = 30
			}
			change_development_level = 1
		}
		add_gold = 100
		add_piety = minor_piety_loss
	}

	option = {
		name = rse_slave.0001.b
		add_character_flag = rse_slave_market_decided
		trigger = {
			faith = {
				OR = {
					religion_tag = christianity_religion
					has_doctrine_parameter = no_same_faith_slaves
				}
			}
		}
		scope:slave_trader = { banish = yes }
		scope:market_county = {
			add_county_modifier = {
				modifier = no_slave_trade_modifier
				years = 15
			}
		}
		add_piety = medium_piety_gain
	}

	option = {
		name = rse_slave.0001.c
		add_character_flag = rse_slave_market_decided
		scope:market_county = {
			add_county_modifier = {
				modifier = regulated_slave_market_modifier
				years = 20
			}
		}
		add_gold = 50
	}
}

# A slave revolt erupts in the domain
rse_slave.0004 = {
	type = character_event
	title = rse_slave.0004.t
	desc = rse_slave.0004.desc
	theme = faith

	left_portrait = {
		character = root
		animation = anger
	}

	trigger = {
		is_landed = yes
		any_held_county = {
			OR = {
				has_county_modifier = slave_market_modifier
				has_county_modifier = regulated_slave_market_modifier
				has_county_modifier = zanj_plantation_modifier
			}
		}
	}

	immediate = {
		random_held_county = {
			limit = {
				OR = {
					has_county_modifier = slave_market_modifier
					has_county_modifier = regulated_slave_market_modifier
					has_county_modifier = zanj_plantation_modifier
				}
			}
			save_scope_as = revolt_county
		}
	}

	option = {
		name = rse_slave.0004.a
		spawn_army = {
			levies = 1500
			men_at_arms = { type = light_footmen stacks = 2 }
			location = scope:revolt_county.title_province
			name = rse_slave_revolt_response
		}
		add_gold = -75
		scope:revolt_county = {
			add_county_modifier = {
				modifier = crushed_slave_revolt_modifier
				years = 8
			}
			change_development_level = -2
		}
		add_dread = major_dread_gain
	}

	option = {
		name = rse_slave.0004.b
		scope:revolt_county = {
			remove_county_modifier = slave_market_modifier
			remove_county_modifier = regulated_slave_market_modifier
			remove_county_modifier = zanj_plantation_modifier
			add_county_modifier = {
				modifier = freedmen_settled_modifier
				years = 15
			}
		}
		add_piety = medium_piety_gain
		add_prestige = minor_prestige_loss
	}

	option = {
		name = rse_slave.0004.c
		scope:revolt_county = {
			add_county_modifier = {
				modifier = slaves_pacified_modifier
				years = 5
			}
		}
		add_gold = -40
	}
}

# A low-born beauty is brought to court
rse_slave.0010 = {
	type = character_event
	title = rse_slave.0010.t
	desc = rse_slave.0010.desc
	theme = faith

	left_portrait = root
	right_portrait = {
		character = scope:lowborn_beauty
		animation = personality_compassionate
	}

	trigger = {
		is_landed = yes
	}

	immediate = {
		create_character = {
			template_character = root
			location = root.capital_province
			faith = root.faith
			culture = root.culture
			gender_female_chance = root.gender_female_inverse_chance
			age = { 16 22 }
			save_scope_as = lowborn_beauty
			employer = root
			random_traits_list = {
				count = 2
				50 = { add_trait = beauty_good_3 }
				50 = { add_trait = gregarious }
				50 = { add_trait = compassionate }
				50 = { add_trait = patient }
			}
			add_trait = rse_low_born_beauty
		}
	}

	option = {
		name = rse_slave.0010.a
		trigger = {
			faith = {
				OR = {
					religion_tag = islam_religion
					has_doctrine_parameter = unrestricted_concubines
				}
			}
		}
		scope:lowborn_beauty = {
			add_trait = rse_concubine_slave
		}
		take_concubine = {
			concubine = scope:lowborn_beauty
		}
		add_piety = minor_piety_loss
	}

	option = {
		name = rse_slave.0010.b
		scope:lowborn_beauty = {
			set_employer = root
			add_opinion = {
				modifier = grateful_opinion
				target = root
				years = 30
			}
		}
		add_prestige = minor_prestige_gain
	}

	option = {
		name = rse_slave.0010.c
		scope:lowborn_beauty = {
			add_trait = rse_enslaved
		}
		scope:lowborn_beauty = {
			add_opinion = {
				modifier = enslaved_opinion
				target = root
				years = 100
			}
		}
		add_gold = -25
	}

	option = {
		name = rse_slave.0010.d
		scope:lowborn_beauty = { banish = yes }
		add_piety = minor_piety_gain
		add_gold = 25
	}
}
